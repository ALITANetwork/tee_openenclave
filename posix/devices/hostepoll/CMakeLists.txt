add_library(oehostepoll STATIC hostepoll.c)

maybe_build_using_clangw(oehostepoll)

    ${CMAKE_BINARY_DIR}/posix

    DESTINATION ${CMAKE_INSTALL_LIBDIR}/openenclave/enclave)
