set (EDL_FILE ../genkey.edl)

add_custom_command(
    OUTPUT genkey_u.h genkey_u.c
    DEPENDS ${EDL_FILE} edger8r
    COMMAND edger8r
        --untrusted ${EDL_FILE}
        --search-path ${CMAKE_CURRENT_SOURCE_DIR})

add_executable(genkey_host host.c genkey_u.c)

target_include_directories(genkey_host PRIVATE ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(genkey_host oehostapp)
